def generate_response(prompt: str, model: str = "gpt-4") -> str:
    """
    Generate a response from the LLM based on the given prompt.
    
    Args:
        prompt: The input prompt text
        model: The model to use for generation
        
    Returns:
        The generated response as a string
    """
    try:
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": "You are a helpful assistant."},
                {"role": "user", "content": prompt}
            ],
            temperature=0.7,
            max_tokens=1000
        )
        return response.choices[0].message.content
    except Exception as e:
        logger.error(f"Error generating response: {e}")
        return None
        
def chain_of_thought(question: str, steps: int = 3) -> str:
    """
    Implement Chain-of-Thought prompting for complex reasoning.
    
    Args:
        question: The complex question requiring multi-step reasoning
        steps: The number of reasoning steps to include
        
    Returns:
        The final answer with reasoning steps
    """
    prompt = f"""
    Question: {question}
    
    Please solve this step-by-step:
    1. Break down the problem
    2. Work through each part sequentially
    3. Combine the results for a final answer
    
    Provide your complete thought process.
    """
    
    response = generate_response(prompt)
    return response
